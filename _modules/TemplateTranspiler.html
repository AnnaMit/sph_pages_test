[docs]    def __init__(self, data_type='float', record_filename='data.codegen.discrete_records.record_1',
                 loop_type='config', code_output_directory=None, output_directory='../data/exec/output'):
        """
        Base constructor for TemplateTranspiler objects. Extends the constructor of base class Transpiler.

        Args:
            data_type (str): Datatype that should be used in the code. Default is 'float'.
            record_filename (types.ModuleType): Module containing the discretized ODE system.
            loop_type (str): Type of the loop used for iteration in implicit integration methods (ignored otherwise).
                Default is 'config'. Other possibilities: 'for', 'both', 'while', 'counter'.
            code_output_directory (str): Path to the directory where the generated code files should be saved.
                Default is None.
            output_directory (str): Path of the directory of the default file for the integrated solution.
                Default is '../data/exec/output'.
        """
        record = self.load_record(record_filename)
        super().__init__(record, output_directory=output_directory, data_type=data_type,
                         code_output_directory=code_output_directory, loop_type=loop_type)
        # File with modules to fill some template properties
        self.module_file = os.path.join(os.path.dirname(__file__), "templates", self.dir_name + "/code_modules.txt")
        self.equation_transpiler = self.equation_transpiler_class()
        self.config = {'num_equations' : len(self.rec.variablelookuptable),
                       'num_parameters': len(self.rec.parameterlookuptable),
                       'variable_names': "\"%s\"" % ",".join(self.rec.variablelookuptable)}

    @property
    @abstractmethod
    def marker(self):
        """
        Marker that indicates a part of the template that needs to be replaced by generated code.

        These parts begin and end with marker.

        Example:
            ``marker = '##'`` causes ``## example ##`` to be replaced by the value of ``config['example']``.
        """
        pass

    @property
    @abstractmethod
    def dir_name(self):
        """
        Name of the subdirectory of ``src/exec`` where the code should be generated, e.g. ``c`` or ``openmp``.
        """
        pass

    @property
    @abstractmethod
    def template_file_dict(self):
        """
        Dictionary of template file paths and the paths and the corresponding paths where target files should be generated.

        return {}

        pass

[docs]    def generate(self):
        """
        Generate code implementing the system specified in ``self.record``.

        This is done  loading code from the template files and replacing the sections marked with ``self.marker``
        with data from ``self.config`` (discretized system) and ``self.module_file``
        """
        self.generate_config()  # Implementation specific data --> config
        self.read_modules()  # Read in data from self.module_file --> config
        self.generate_equations()  # Generate the code for the equations --> config
        self.generate_local_variables()  # Generate code for declaration of local variables --> config
        self._generate_all_files()  # Read template and replace marked parts by the corresponding items in config

[docs]    def generate_equations(self):
        """
        Add an item to ``self.config`` that contains code implementing the equations of discretized system.
        """
        # 1) Generate code that updates the variable values
        str_mainloop = "\n"
        for curr_eq in self.rec.disc_equations:
            # type is while --> this is the loop approximating the step solutions in implicit methods
            if type(curr_eq) == sp.codegen.ast.While:
                str_mainloop += self.implement_approximation_loop(curr_eq)
            # Every other type should be just a normal equation
            else:
                str_mainloop += self.equation_transpiler.generate_assignment(curr_eq)

        # 2) Generate code for printing the results to the command line
        if self.print_results:
            self.generate_print_code()
            str_mainloop += self.generate_print_code()
        self.config['equations'] = str_mainloop

        """
        Generate code for declaring the local variables and arrays for the implicit Runge-Kutta stages.
        """
        # Most of the code taken from c_transpiler2.py
        # Set up a local variable for every RungeKutta Stage, avoid double declaration
        # Create the list of all equation.lhs
        # Set up a local variable for every RungeKutta Stage, avoid double declaration
        # Create the list of all equation.lhs
        equality_list = [l for l in self.rec.disc_equations if (type(l) == sp.Equality)]
        while_list = [l for l in self.rec.disc_equations if (type(l) == sp.codegen.ast.While)]
        while_equality_list = [item for l in while_list for item in l.body._sorted_args]

        lhs_list = list(map(lambda l: l.lhs, equality_list))
        lhs_list += list(map(lambda l: l.lhs, while_equality_list))

        # Eliminate duplicates
        lhs_list = list(set(lhs_list))

        # Write declarations to the ccode
        str_locvars = "\n"
        for curr_lhs in lhs_list:
            str_locvars += self.declare_local_variable(curr_lhs)

        # implicit methods array for RK stages
        if self.rec.iterative_method == "True":
            RK_arrays = list(set(str(a).split("[")[0] for a in lhs_list if '[' in str(a) and str(a).startswith("RK")))
            for a in RK_arrays:
                str_locvars += self.declare_RK_array(a)

        self.config["local variables"] = str_locvars

[docs]    def adapt_line(self, line, records):
        """
        Adapt a line.
        """
        if not (self.line_needs_adaptment(line)):
            return line

        variable_name = line.split("=")[0].strip()
        array_variables = {str(a): b for (a, b) in records.timestep_specific_variables.items()}
        if variable_name in array_variables:
            if not hasattr(self, "adapt_variable_name"):
                return line
            time_offset = array_variables[variable_name][1]
            index = array_variables[variable_name][0]
            return self.adapt_variable_name(index, time_offset, variable_name)
        elif variable_name in records.parameterlookuptable:
            if not hasattr(self, "adapt_parameter_name"):
                return line
            return self.adapt_parameter_name(records.parameterlookuptable.index(variable_name), variable_name)

        elif records.iterative_method == "True":
            if not hasattr(self, "adapt_backwardRK_variables"):
                return line
            return self.adapt_backwardRK_variables(variable_name, line)
        else:
            return line



[docs]    def line_needs_adaptment(self, line):
        # If line is no form of ... = Symbol(... don't change anything
        return len(line.split("= Symbol(")) == 2 or len(line.split("=Symbol(")) == 2

[docs]    def load_record(self, record_filename):
        """
        Adapt and load the module file containing the discretized ODE system.

        Calls ``adapt_records`` to adapt the record file for the specific transpiler and reloads the records.

        Args:
            record_filename (str): Path of the module file containing the discretized ODE system.
        Returns:
             types.ModuleType: The adapted and loaded record module.
        """
        # Load record file with the discretized equations
        record = import_module_from_path(record_filename)
        # Adapt the records to the specific language and save them as a new record file.
        self.adapt_records(record, record_filename)
        new_record = import_module_from_path(self.adapted_record_file)
        return new_record



[docs]    def adapt_records(self, records, record_file):
        """
        Creates a version of the record file with changes for code generation in the corresponding language.

        Args:
            records (types.ModuleType): The module containing the original record.
            record_file (str): Path of the original record file.
        """
        with open(record_file) as file:
            lines = file.readlines()

        text = []
        for line in lines:
            line = line.strip()
            text.append(self.adapt_line(line, records))

        text = '\n'.join(map(str, text))
        print_content_to_file(self.adapted_record_file, text)


    ##### Used for code generation #####
    def _process_code_line(self, line):
        """
        Replace all marked parts of ``line``, use the correct datatype and return the resulting line.
        """
        line = self._replace_line_modules(line)
        line = self._replace_variable_modules(line)
        line = self.correct_datatype(line)
        return line

    def _replace_line_modules(self, line):
        """
        If the whole ``line`` is marked, return the corresponding item in ``self.config`` otherwise return ``line``.
        """
        stripped_line = line.strip()
        if self.line_is_marker(stripped_line):
            try:
                line = self.config[self.unmark_string(stripped_line)]

            # If there is no entry for this line (or any other error), show an error message and don't replace anything.
            except:
                print_error(
                        'Problem during code generation. There is no value for the key {0}. Not replacing the line.'.format(
                                stripped_line))
                line = line
        return line

    def _replace_variable_modules(self, line):
        """
        Insert the variables in ``self.config`` at the marked locations in line.
        """
        for variable in self.config:
            line = line.replace(self.marked_string(variable), str(self.config[variable]))
        return line

    def _read_code_from_template(self, filepath):
        """
        Returns the processed content of the code file (after replacing marked variables etc.).
        """
        with open(filepath) as file:
            lines = file.readlines()
            lines = list(map(self._process_code_line, lines))
        return lines

    ##### I/O related methods
    def _generate_all_files(self):
        """
        Copies template files to target location without changing their content.
        """
        template_file_dict_paths = self._get_all_filepaths()
        for file in template_file_dict_paths:
            code = self._read_code_from_template(file)
            print_content_to_file(template_file_dict_paths[file], code)

    def _get_all_filepaths(self):
        """
        Return the relative paths to all template and target files.

         Generate the path to all template and target files in ``self.template_file_dict``:
            * The parent directory for the target files is ``self.code_output_directory``. If this is ``None``, the default target location is ``src/exec/<self.dir_name>``.
            * The parent directory for template files is ``src/codegen/templates/<self.dir_name>``.
        """
        # Current directory
        codegen_dir = os.path.dirname(__file__)

        template_file_dict_paths = {}
        # Put all paths to template files and target files i the dict
        for file in self.template_file_dict:
            target_file = self.template_file_dict[file]
            if self.code_output_directory is not None:
                target_file = os.path.join(self.code_output_directory, target_file)
            else:
                target_file = os.path.join(os.path.dirname(codegen_dir), "exec", self.dir_name, target_file)

            template_file = os.path.join(codegen_dir, "templates", self.dir_name, file)
            template_file_dict_paths[template_file] = target_file
        return template_file_dict_paths


[docs]    def read_modules(self):
        """
        Read module file and put markers and their data in the module dictionary ``self.config``.
        """
        config = configparser.ConfigParser(comment_prefixes=('%%', '%%'), inline_comment_prefixes=None)
        config.read(self.module_file)
        for key in config['DEFAULT']:
            self.config[key] = config['DEFAULT'][key]

[docs]    def unmark_string(self, marked_string):
        """
        Remove marker from ``marked_string``.
        """
        return marked_string.replace(self.marker, '')



[docs]    def marked_string(self, s):
        """
        Append markers to a string ``s``.
        """
        return "%s%s%s" % (self.marker, s, self.marker)
