[docs]def make_all_tests(dt, num_steps, t0, num_steps_comparison=-1):
    for system in systems:
        config_file = ("data.disc.example_configurations.%s.config_%s" % (system, system))
        input_file = "../tests/grid_test_input/test_input_%s.csv" % system

        make_multiple_test_inputs(input_file.strip("../"), 8)

        path = os.path.join(directory_path, system)
        if not os.path.exists(path):
            os.mkdir(path)

        ex = importlib.import_module(config_file)
        importlib.reload(ex)
        num_eq = len(ex.variablenames.split(" "))

        for method in methods:
            # Discretization only needed once for all languages
            path_m = os.path.join(path, method)
            if not os.path.exists(path_m):
                os.mkdir(path_m)
            method_list = [method] * num_eq
            start = time.process_time()
            discretize(discrete_records_filename="data/codegen/discrete_records/debug_record.py",
                       configuration_filename=config_file, methods=method_list)
            end = time.process_time()
            print("Discretization of %s model with %s method took %s seconds" % (system, method, str(end - start)))
            if method in implicit:
                loops = loop_types
            else:
                loops = ["x"]
            for loop_type in loops:
                path_t = os.path.join(path_m, loop_type)
                if not os.path.exists(path_t):
                    os.mkdir(path_t)
                for language in languages:
                    path_l = os.path.join(path_t, language.lower())
                    if not os.path.exists(path_l):
                        os.mkdir(path_l)
                    test_case(dt, num_steps, t0, language, method, config_file, input_file, system, path_l,
                              loop_type=loop_type)




[docs]def make_multiple_test_inputs(input_file, num_lines):
    with open(input_file) as f:
        line = f.readline().strip().split(";")
        init_vals = line[0].strip().split(" ")
        init_vals = list(map(lambda x: float(x), init_vals))
        para_vals = line[1].strip().split(" ")
        para_vals = list(map(lambda x: float(x), para_vals))

    offset_i = init_vals[0] / 1000
    offset_p = para_vals[0] / 1000
    with open(input_file, "w") as f:
        for i in range(num_lines):
            print(" ".join(map(lambda x: str(x), init_vals)), " ; ", " ".join(map(lambda x: str(x), para_vals)), file=f)
            init_vals[0] += offset_i
            para_vals[0] += offset_p
