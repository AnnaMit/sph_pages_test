[docs]    def annotate_TIU(self, methods, variables):
        """
        This method annotates an instance of ODE with the prior (i.e in config.py) specified timeintegration methods.
        Returns:
            An instance of TimeIntegratorAnnotatedODE corresponding to this instance of ODE and the prior specified timeintegration methods.
        """
        annot_equations = ()
        for i in range(len(self.equations)):
            curr_eq = self.equations[i]

            curr_method = methods[i].split()[0]


            # Create a dummy class instance of the requested method, to decide how to initialize the TimeIntegratorUnit for this equation
            dummy = tiu.annotation_dict[curr_method]


            # Case method is an operator splitting technique
            if issubclass(dummy, tiu.OperatorSplitting):

                #Split the whole equation system into first addend and remaining addends
                split_ode1 = ODE(())
                split_ode2 = ODE(())
                for eq in self.equations:
                    # Case an equation is not an addition and thus is split into eq.rhs + 0
                    if eq.rhs.func != sp.Add:
                        split_ode1.app(eq)
                        split_ode2.app(sp.Eq(eq.lhs, 0))
                    else:
                        split_ode1.app(sp.Eq(eq.lhs, eq.rhs.args[0]))  # The first addend of each equation goes to the first split
                        split_ode2.app(sp.Eq(eq.lhs, sum(eq.rhs.args[1:])))  # The remaining addeds go to the second split

                # Annotated each equation with the methods specified in methods[i].split[1:2]
                # Retrieve The methods, how to integrate the split terms
                tiu1_method = methods[i].split()[1]
                tiu2_method = methods[i].split()[2]
                split_annot1 = split_ode1.annotate_TIU(((tiu1_method,) * len(split_ode1.equations)) , variables)
                split_annot2 = split_ode2.annotate_TIU(((tiu2_method,) * len(split_ode2.equations)) , variables)
                #Create the OperatorSplitting object
                annot_list = [split_annot1, split_annot2]
                opsplit = tiu.annotation_dict[curr_method](annot_list, variables)
                annot_equations += (opsplit, )
                #Break out as the whole system will be integrated with the splitting method now
                break

            # Case method that creates a new system of equations to solve for several at the same time
            elif issubclass(dummy, tiu.DependentSystemTimeIntegratorUnit):
                # Check all previously added equations if one is already a DependentSystemTimeIntegratorUnit
                # If yes, don't make a new object as the first one already contains the new one
                annot_equations = self.annotate_dependent_system(annot_equations, curr_eq, curr_method, variables)

            # Case method that respects dependencies between equations (i.e. RungeKutta)
            elif issubclass(dummy, tiu.MultiDependencyTimeIntegratorUnit):
                terms = ()
                terms_to_traverse = (curr_eq,)
                while len(terms_to_traverse) != 0:
                    terms += terms_to_traverse
                    new_terms = ()
                    for curr_term in terms_to_traverse:
                        for arg in sp.preorder_traversal(curr_term):
                            # print(arg, arg.func)
                            # Case we are at a variable (i.e. x_i(t)) and its not yet on the lhs of the equations in the terms tuple or new_terms tuple
                            if type(arg.func) == sp.function.UndefinedFunction and not arg in list(
                                    map(lambda x: x.lhs.args[0], terms)) and not arg in list(
                                    map(lambda x: x.lhs.args[0], new_terms)):
                                # Traverse self.equations and find the ODE for the derivative of the just found variable
                                for curr_eq in self.equations:
                                    if curr_eq.lhs.args[0] == arg:
                                        new_terms += (curr_eq,)
                    # Update the terms to traverse in the next search
                    terms_to_traverse = new_terms

                    # Create the MultiDependencyTimeIntegratorUnit object
                annot_equations += (tiu.annotation_dict[curr_method](terms, variables),)

            # Case standard integration method type (i.e. ExplicitEuler)
            # This case requires issubclass(dummy, tiu.TimeIntegratorUnit) == True
            else:
                annot_equations += (tiu.annotation_dict[curr_method](curr_eq, variables), )

        return TimeIntegratorAnnotatedODE(annot_equations)



[docs]    def annotate_dependent_system(self, annot_equations, curr_eq, curr_method, variables):
        """
        Create a DependentSystemTimeIntegratorUnit instance for an equality.

        Args:
            annot_equations (tuple of src.odedsl.TimeIntegratorUnit):  Already annotated equations. This method extends this tuple.
            curr_eq (sympy.core.relational.Equality): Current equality to annotate.
            curr_method (str):  Integration method for the current equality.
            variables (tuple of sympy.core.symbols.Symbol): All variables in the equation (left hand side of ODEs).

        Returns:
            src.odedsl.TimeIntegratorUnit.TimeIntegratorUnit: TimeIntegratorUnit instance for the given equality and method.

        """
        for previous_equation in annot_equations:
            if issubclass(type(previous_equation), tiu.DependentSystemTimeIntegratorUnit):
                return annot_equations
        terms = ()
        terms_to_traverse = (curr_eq,)
        while len(terms_to_traverse) != 0:
            terms += terms_to_traverse
            new_terms = ()
            for curr_term in terms_to_traverse:
                for arg in sp.preorder_traversal(curr_term):
                    # Case we are at a variable (i.e. x_i(t)) and its not yet on the lhs of the equations in the terms tuple or new_terms tuple
                    if type(arg.func) == sp.function.UndefinedFunction and not arg in list(
                            map(lambda x: x.lhs.args[0], terms)) and not arg in list(
                            map(lambda x: x.lhs.args[0], new_terms)):
                        # Traverse self.equations and find the ODE for the derivative of the just found variable
                        for curr_eq in self.equations:
                            if curr_eq.lhs.args[0] == arg:
                                new_terms += (curr_eq,)
            # Update the terms to traverse in the next search
            terms_to_traverse = new_terms
        # Create the MultiDependencyTimeIntegratorUnit object
        annot_equations += (tiu.annotation_dict[curr_method](terms, variables),)
        return annot_equations



[docs]    def equations_canonic(self, orig_variables, variables):
        """
        Substitute all variable in all equations with their corresponding canonic form.

        Args:
            orig_variables: The tuple of variables in original form in dependence of t.
            variables: The tuple of variables in canonic form in dependence of t.
        """
        t = sp.Symbol("t")
        equations_with_xi = list(self.equations)
        # substitute all symbols in the original ode with the corresponding x_i
        for i in range(len(variables)):
            for j in range(len(self.equations)):
                if type(orig_variables) == tuple:
                    equations_with_xi[j] = equations_with_xi[j].subs(orig_variables[i](t), variables[i])
                # else @variables and @orig_variables are only symbols and no tuples
                else:
                    equations_with_xi[j] = equations_with_xi[j].subs(orig_variables(t), variables[0])

        equations_with_xi = tuple(equations_with_xi)
        self.set_equations(equations_with_xi)

[docs]    def transform_ODE(self):
        """
        Transform an instance of TimeIntegratedAnnotatedODE into a corresponding instance of ODE.

        Returns:
            An instance of ODE.
        """
        # trans_equations is the placeholder tuple, which will be extended and used to create an instance of ODE.
        trans_equations = ()
        t = sp.Symbol("t")
        for curr_tiu in self.equations:
           
            if issubclass(type(curr_tiu), tiu.OperatorSplitting):
                #Transform the two split up annotated systems into ODE objects
                help_ode1 = curr_tiu.annot1.transform_ODE()
                help_ode2 = curr_tiu.annot2.transform_ODE()
                
                #Create the complete system by adding the rhs of the two helper ODEs
                for i in range(len(help_ode1)):
                    trans_equations += (sp.Eq( help_ode1.get_equations()[i].lhs , help_ode1.get_equations()[i].rhs + help_ode2.get_equations()[i].rhs ), )
                

            #Case only a single time integration method
            elif issubclass(type(curr_tiu),tiu.MultiDependencyTimeIntegratorUnit):
                trans_equations += (sp.Eq(curr_tiu.lhs_list[0].diff(t), curr_tiu.terms[0]), )
            else:
                trans_equations += (sp.Eq(curr_tiu.lhs.diff(t), curr_tiu.term), )


        return ODE(trans_equations)



[docs]    def transform_TIES(self):
        """
        This method is used to transform an instance of TimeIntegratedAnnotatedODE into a corresponding instance of TimeIntegratedEquationSystem.

        Returns:
            An instance of TimeIntegratedEquationSystem consisting of and equations !*LIST*! corresponding to the original ODE with applied timeintegration methods.
        """

        ties = TimeIntegratedEquationSystem(())
        for curr_tiu in self.equations:
            curr_tiu.resolve(ties)

        return ties

[docs]    def subs_deltaT(self, dT):
        """
        Substitutes the sympy variable deltaT with the passed Integervalue ``dT``.
        
        Args:
            dT (float): The value to substitute deltaT with.
        """
        new_equations = ()
        deltaT = sp.Symbol("deltaT")
        for curr_eq in self.equations:
            new_equations += ( curr_eq.subs(deltaT, dT), )

        self.set_equations(new_equations)
